442 score += 1;

448 } else {

450 score += (1- std::abs(item.height - p.height) / params.printable_height)

452 score += LARGE_COST_TO_REJECT * (item.bed_temp - p.bed_temp != 0);
453 }
454 }

473 }

475 // this has very weak effect, CAN NOT use a large weight
476 extruder_ids.insert(item.extrude_ids.begin(), item.extrude_ids.end());
477 if (!params.is_seq_print) {
478 score += 1 * std::max(0, ((int) extruder_ids.size() - 1));
479 }
